chrout = $ffd2
ptr    = $fb
count  = $fd
       lda #2
       sta count
       ldx #$f4
       lda #<msg
       sta ptr
       lda #>msg
       sta ptr+1
outer: ldy #0
inner: lda (ptr),y
       beq done
       jsr chrout
       iny
       bne inner
done:  dex
       bne outer
       dec count
       bne outer
       rts
msg: .byte 14,"I will not throw paper airplanes in class.",13,0
