% some convenience predicates (these are predefined in SWI-Prolog)
foreach(Generator, Goal) :-
    findall(Goal, Generator, Goals),
    maplist(call, Goals).

writeln(Goal) :-
    write(Goal),
    nl.

main :-
    foreach(between(1,500,_),
    writeln('I will not throw paper airplanes in class.')), halt.
